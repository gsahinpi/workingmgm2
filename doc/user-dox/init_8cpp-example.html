#include <new>
#include <iostream>
#include "init.h"
extern Magmio::MagmioApi *api;
extern std::atomic<bool> stop;
void fwInterruptHandler(const std::string reason) {
std::cerr << "Interrupt Handler: Interrupt received: " << reason << std::endl;
stop.store(true, std::memory_order_release);
if (api) {
api->abort();
std::cout << "Interrupt Handler: API abort has finished." << std::endl;
}
}
void swExceptionHandler(const std::exception_ptr p_exception) {
try {
std::rethrow_exception(p_exception);
}
catch (const std::exception &e) {
std::cerr << "Exception Handler: Exception received: " << e.what() << std::endl;
stop.store(true, std::memory_order_release);
if (api) {
api->abort();
std::cout << "Exception Handler: API abort has finished." << std::endl;
}
}
}
Magmio::TError assignCPUCores(Magmio::MagmioApi &api) {
cpu_set_t cpuSet;
CPU_ZERO(&cpuSet);
// Default: Use all available cpu cores
for (unsigned int i = 0; i < CPU_SETSIZE; i++) {
CPU_SET(i, &cpuSet);
}
// Affinity must be set before api initialization
// Set affinity for internal API status channel thread
return api.setAffinity(Magmio::Settings::THR_ID_STATUS, &cpuSet);
}
Magmio::TError create(Magmio::MagmioApi **outApi, int cardIndex, const std::string fileConfig, Magmio::Settings::TApiExtras settings) {
try {
// Try to create a new API object
*outApi = new Magmio::MagmioApi(cardIndex, fileConfig, settings);
} catch (const std::bad_alloc &e) {
*outApi = NULL;
std::cerr << "ERROR: Creating API object failed!" << std::endl;
return Magmio::MEMORY_ERROR;
}
return Magmio::SUCCESS;
}
Magmio::TError initialize(Magmio::MagmioApi &api, const std::string fileFirmware, const std::string fileMapping,
char* fileStrategy, char* fileStrategyPrefilter, const Magmio::Log::tVerbosityLevel screenVerbosity,
const Magmio::Log::tVerbosityLevel logVerbosity, const std::string logFilename,
char *fileMcastRecord, char *fileOrderRecord, char *extraParamsFile) {
Magmio::TError error;
// Set user interrupt callback when FW interrupt occurs
api.setInterruptCallback(std::bind(fwInterruptHandler, std::placeholders::_1));
// Set user exception callback when SW exception occurs
api.setExceptionCallback(std::bind(swExceptionHandler, std::placeholders::_1));
api.log.setPrintVerbosity(screenVerbosity);
if (logVerbosity > Magmio::Log::QUIET) {
if ((error = api.log.createFile(logVerbosity, logFilename.c_str())) != Magmio::SUCCESS) {
std::cerr << "ERROR (Initialization): Creating log file failed!" << std::endl;
return error;
}
}
// Record orders sent by FPGA
if (fileOrderRecord) {
if ((error = api.log.createOrderPcap(fileOrderRecord)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Initialization): Creating orders pcap file failed!" << std::endl;
return error;
}
}
if ((error = assignCPUCores(api)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Initialization): Affinity setup failed!" << std::endl;
return error;
}
// Initialize API
if ((error = api.init(fileFirmware, fileMapping)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Initialization): API init failed!" << std::endl;
return error;
}
// Check strategy consistency
if (fileStrategy) {
if ((error = api.verifyStrategyConsistency(fileStrategy)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Initialization): Strategy verification failed!" << std::endl;
return error;
}
}
// Load strategy prefilter configuration from file
if (fileStrategyPrefilter) {
if ((error = api.strategyPrefilter.loadFromFile(fileStrategyPrefilter)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Initialization): Loading strategy prefilter configuration from file failed!" << std::endl;
return error;
}
}
// Record mcast traffic received by FPGA
if (fileMcastRecord) {
if ((error = api.stream.pcap.record(fileMcastRecord)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Initialization): Opening multicast traffic pcap file failed!" << std::endl;
return error;
}
}
// Load extra params from file
if (extraParamsFile) {
if ((error = api.taker.loadExtraFromFile(extraParamsFile)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Initialization): Loading extra parameters from file failed!" << std::endl;
return error;
}
}
return Magmio::SUCCESS;
}
