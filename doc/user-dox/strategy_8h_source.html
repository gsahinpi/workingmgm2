167  FeedbackMessage(const FeedbackMessage &other) {
168  timestamp = other.timestamp;
169  if ((other.size) && (other.data) && (data = (uint8_t *)malloc(other.size))) {
170  memcpy(data, other.data, other.size);
171  size = other.size;
172  index = other.index;
173  } else {
174  data = nullptr;
175  size = 0;
176  index = 0;
177  }
178  }
179 
180  FeedbackMessage& operator=(const FeedbackMessage &other) {
181  timestamp = other.timestamp;
182  if (data) {
183  free(data);
184  }
185  if ((other.size) && (other.data) && (data = (uint8_t *)malloc(other.size))) {
186  memcpy(data, other.data, other.size);
187  size = other.size;
188  index = other.index;
189  } else {
190  data = nullptr;
191  size = 0;
192  index = 0;
193  }
194  return *this;
195  }
