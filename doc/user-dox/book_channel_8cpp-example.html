#include <iostream>
#include "book_channel.h"
Magmio::TError processBookMessage(std::ofstream &out, Magmio::BookChannel::ExportMessage &msg) {
// Print message
out << msg << std::endl;
return Magmio::SUCCESS;
}
void runBookChannel(std::promise<bool> &&initStatus, std::promise<Magmio::TError> &&exitStatus, std::atomic<bool> &stop, Magmio::MagmioApi &api, cpu_set_t cpuSet) {
Magmio::TError setupError {Magmio::SUCCESS};
Magmio::TError processError {Magmio::SUCCESS};
Magmio::BookChannel::ExportMessage msg;
// Set thread's affinity
int retcode;
if ((retcode = pthread_setaffinity_np(pthread_self(), sizeof(cpu_set_t), &cpuSet)) != 0) {
std::cerr << "ERROR (Book Channel): Unable to set desired thread affinity - " << strerror(retcode) << std::endl;
initStatus.set_value(false);
exitStatus.set_value(Magmio::OTHER_ERROR);
return;
}
while (!api.isInitialized()) {
if (stop.load(std::memory_order_acquire)) {
std::cerr << "ERROR (Book Channel): API not initialized!" << std::endl;
initStatus.set_value(false);
exitStatus.set_value(Magmio::NOT_INITIALIZED);
return;
}
std::this_thread::sleep_for(std::chrono::milliseconds(100));
}
// Create file for stream output
std::ofstream outFile("book.out", std::ofstream::out | std::ofstream::trunc);
if (!outFile.is_open()) {
std::cerr << "ERROR (Book Channel): Output file opening failed!" << std::endl;
initStatus.set_value(false);
exitStatus.set_value(Magmio::OPEN_FILE);
return;
}
// Subscribe to receive book channel messages
if ((setupError = api.stream.book.subscribe()) != Magmio::SUCCESS) {
std::cerr << "ERROR (Book Channel): " << Magmio::Log::getErrorMessage(setupError) << std::endl;
outFile.close();
initStatus.set_value(false);
exitStatus.set_value(std::move(setupError));
return;
}
initStatus.set_value(true);
// Read messages from book channel in infinite loop (until stopped or failed)
while (!stop.load(std::memory_order_acquire)) {
// Read message
processError = api.stream.book.getNextMessage(msg);
if (processError == Magmio::STREAM_INTERRUPTED) {
// Interrupted stream
break;
} else if (processError != Magmio::SUCCESS) {
std::cerr << "ERROR (Book Channel): " << Magmio::Log::getErrorMessage(processError) << std::endl;
break;
}
// Process message
if ((processError = processBookMessage(outFile, msg)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Book Channel): Processing message failed!" << std::endl;
break;
}
}
// Unsubscribe from book channel at the end
if ((setupError = api.stream.book.unsubscribe()) != Magmio::SUCCESS) {
std::cerr << "ERROR (Book Channel): " << Magmio::Log::getErrorMessage(setupError) << std::endl;
}
outFile.close();
exitStatus.set_value((processError != Magmio::SUCCESS) ? std::move(processError) : std::move(setupError));
}
