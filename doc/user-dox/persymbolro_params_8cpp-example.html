#include <iostream>
#include "persymbolro_params.h"
Magmio::TError testPerSymbolRoParams(Magmio::MagmioApi &api, const uint64_t sid) {
Magmio::TError error;
Magmio::PerSymbolRoParams::tParam param;
Magmio::PerSymbolRoParams::tParams params;
const unsigned int perSymbolRoParamsCount = api.params.persymbolro.getNumParams();
if (!api.isInitialized()) {
return Magmio::NOT_INITIALIZED;
}
if (perSymbolRoParamsCount == 0) {
std::cout << "Persymbolro Params: None of params are available for symbol " << sid << "." << std::endl;
return Magmio::SUCCESS;
}
// Set parameter 0 to -1000
std::cout << "Persymbolro Params: Set parameter 0 for symbol " << sid << std::endl;
if((error = api.params.persymbolro.setParam(sid, Magmio::PerSymbolRoParams::tParam(0, -1000))) != Magmio::SUCCESS) {
std::cerr << "ERROR (Persymbolro Params): Setting parameter 0 failed!" << std::endl;
return error;
}
// Get parameter 0
std::cout << "Persymbolro Params: Get parameter 0 for symbol " << sid << std::endl;
// Parameter 0
param.first = 0;
// Initialize value to 0. This value will be rewritten
param.second = 0;
if ((error = api.params.persymbolro.getParam(sid, param)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Persymbolro Params): Getting parameter 0 failed!" << std::endl;
return error;
} else {
std::cout << "Persymbolro Params: symbol " << sid << " parameter 0 = " << (int)param.second << std::endl;
}
// The next tests require at least two PerSymbolRo params
if (perSymbolRoParamsCount < 2) {
return Magmio::SUCCESS;
}
// Set first and last parameters at once (faster)
std::cout << "Persymbolro Params: Set first and last parameter for symbol " << sid << std::endl;
params.clear();
params.push_back(Magmio::PerSymbolRoParams::tParam(0, 1111));
params.push_back(Magmio::PerSymbolRoParams::tParam(perSymbolRoParamsCount-1, 2222));
if ((error = api.params.persymbolro.setParams(sid, params)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Persymbolro Params): Setting parameters failed!" << std::endl;
return error;
}
// Get first and last parameters at once (faster)
std::cout << "Persymbolro Params: Get first and last parameter for symbol " << sid << std::endl;
params.clear();
params.push_back(Magmio::PerSymbolRoParams::tParam(0, 0));
params.push_back(Magmio::PerSymbolRoParams::tParam(perSymbolRoParamsCount-1, 0));
if ((error = api.params.persymbolro.getParams(sid, params)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Persymbolro Params): Getting parameters failed!" << std::endl;
return error;
}
for (auto item : params) {
std::cout << "Persymbolro Params: symbol " << sid << " parameter " << (int)item.first << " = " << (int)item.second << std::endl;
}
// Get all parameters at once
std::cout << "Persymbolro Params: Get all parameters at once for symbol " << sid << std::endl;
params.clear();
// If no params are specified in vector (empty), all parameters are get
if ((error = api.params.persymbolro.getParams(sid, params)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Persymbolro Params): Getting all parameters failed!" << std::endl;
return error;
}
for (auto item : params) {
std::cout << "Persymbolro Params: symbol " << sid << " parameter " << (int)item.first << " = " << (int)item.second << std::endl;
}
// Set all paremeters at once
std::cout << "Persymbolro Params: Set all parameters at once for symbol " << sid << std::endl;
params.clear();
for (unsigned int i = 0; i < perSymbolRoParamsCount; i++) {
params.push_back(Magmio::PerSymbolRoParams::tParam(i, 1));
}
if ((error = api.params.persymbolro.setParams(sid, params)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Persymbolro Params): Setting all parameters failed!" << std::endl;
return error;
}
// Get all parameters at once
std::cout << "Persymbolro Params: Get all parameters at once for symbol " << sid << std::endl;
params.clear();
// If no params are specified in vector (empty), all parameters are get
if ((error = api.params.persymbolro.getParams(sid, params)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Persymbolro Params): Getting all parameters failed!" << std::endl;
return error;
}
for (auto item : params) {
std::cout << "Persymbolro Params: symbol " << sid << " parameter " << (int)item.first << " = " << (int)item.second << std::endl;
}
return Magmio::SUCCESS;
}
