Detailed Description

Sender pergroup parameters. This class provides API for reading/writing order pergroup parameters.

This file shows how to correctly work with order pergroup parameters.

#include <iostream>
#include "pergroup_params.h"
Magmio::TError testPerGroupParams(Magmio::MagmioApi &api, const uint64_t id) {
Magmio::TError error;
Magmio::PerGroupParams::tParam param;
Magmio::PerGroupParams::tParams params;
const unsigned int pergroupParamsCount = api.params.pergroup.getNumParams();
if (!api.isInitialized()) {
return Magmio::NOT_INITIALIZED;
}
if (pergroupParamsCount == 0) {
std::cout << "Pergroup Params: None of pergroup params are available for Group id " << id << "." << std::endl;
return Magmio::SUCCESS;
}
// Set parameter 0 to -1000
std::cout << "Pergroup Params: Set pergroup parameter 0 for Group id " << id << std::endl;
if((error = api.params.pergroup.setParam(id, Magmio::PerGroupParams::tParam(0, -1000))) != Magmio::SUCCESS) {
std::cerr << "ERROR (Pergroup Params): Setting parameter 0 failed!" << std::endl;
return error;
}
// Get parameter 0
std::cout << "Pergroup Params: Get pergroup parameter 0 for Group id " << id << std::endl;
// Parameter 0
param.first = 0;
// Initialize value to 0. This value will be rewritten
param.second = 0;
if ((error = api.params.pergroup.getParam(id, param)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Pergroup Params): Getting parameter 0 failed!" << std::endl;
return error;
} else {
std::cout << "Pergroup Params: Group id " << id << " Parameter 0 = " << (int)param.second << std::endl;
}
// The next tests require at least two pergroup params
if (pergroupParamsCount < 2) {
return Magmio::SUCCESS;
}
// Set first and last parameters at once (faster)
std::cout << "Pergroup Params: Set first and last pergroup parameters for Group id " << id << std::endl;
params.clear();
params.push_back(Magmio::PerGroupParams::tParam(0, 1111));
params.push_back(Magmio::PerGroupParams::tParam(pergroupParamsCount-1, 2222));
if ((error = api.params.pergroup.setParams(id, params)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Pergroup Params): Setting parameters failed!" << std::endl;
return error;
}
// Get first and last pergroup parameters at once (faster)
std::cout << "Pergroup Params: Get first and last pergroup parameters for Group id " << id << std::endl;
params.clear();
params.push_back(Magmio::PerGroupParams::tParam(0, 0));
params.push_back(Magmio::PerGroupParams::tParam(pergroupParamsCount-1, 0));
if ((error = api.params.pergroup.getParams(id, params)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Pergroup Params): Getting parameters failed!" << std::endl;
return error;
}
for (auto item : params) {
std::cout << "Pergroup Params: Group id " << id << " Parameter " << (int)item.first << " = " << (int)item.second << std::endl;
}
// Get all parameters at once
std::cout << "Pergroup Params: Get all pergroup parameters at once for Group id " << id << std::endl;
params.clear();
// If no params are specified in vector (empty), all parameters are get
if ((error = api.params.pergroup.getParams(id, params)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Pergroup Params): Getting all parameters failed!" << std::endl;
return error;
}
for (auto item : params) {
std::cout << "Pergroup Params: Group id " << id << " Parameter " << (int)item.first << " = " << (int)item.second << std::endl;
}
// Set all paremeters at once
std::cout << "Pergroup Params: Set all pergroup parameters at once for Group id " << id << std::endl;
params.clear();
for (unsigned int i = 0; i < pergroupParamsCount; i++) {
params.push_back(Magmio::PerGroupParams::tParam(i, 1));
}
if ((error = api.params.pergroup.setParams(id, params)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Pergroup Params): Setting all parameters failed!" << std::endl;
return error;
}
// Get all parameters at once
std::cout << "Pergroup Params: Get all pergroup parameters at once for Group id " << id << std::endl;
params.clear();
// If no params are specified in vector (empty), all parameters are get
if ((error = api.params.pergroup.getParams(id, params)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Pergroup Params): Getting all parameters failed!" << std::endl;
return error;
}
for (auto item : params) {
std::cout << "Pergroup Params: Group id " << id << " Parameter " << (int)item.first << " = " << (int)item.second << std::endl;
}
return Magmio::SUCCESS;
}
