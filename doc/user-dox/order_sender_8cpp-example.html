#include <iostream>
#include <chrono>
#include <thread>
#include "order_sender.h"
void watchdog(std::atomic<bool> &stop, Magmio::MagmioApi &api, const uint16_t msTimeout) {
while(!stop.load(std::memory_order_acquire)) {
api.order.wdt_reset();
std::this_thread::sleep_for(std::chrono::milliseconds(msTimeout));
}
api.order.wdt_stop();
}
Magmio::TError processOrderEvent(std::ofstream &out, Magmio::TEventItem &event) {
out << "Received event from order sender " << event.sessionID << " (port #" << event.physicalPort << ")" << std::endl;
// Process event based on its type
switch (event.type) {
case Magmio::TEventType::EVT_ORDER_SEND: {
out << "ORDER #" << event.order.orderID << " SENT\n";
out << "Price: " << std::setprecision(10) << event.order.price << "\n";
out << "Size: " << event.order.size << "\n";
out << "Min QTY: " << event.order.minimumQty << "\n";
out << "Side: " << event.order.side << "\n";
out << "Time in force: " << event.order.timeInForce << "\n";
out << "Order type: " << event.order.orderType << "\n";
out << "Display: " << event.order.displayType << "\n";
out << "ISO: " << event.order.iso << "\n";
out << "Symbol: " << event.order.symbol << "\n";
out << "Symbol (num): " << event.order.symbolNum << "\n";
out << "SID: " << event.order.sid << "\n";
out << "Token: " << event.order.symbolNum << "\n";
out << "Strike price: " << event.order.strikePrice << "\n";
out << "Strike date: " << event.order.strikeDate << "\n";
out << "PutCall: " << event.order.putCall << "\n";
out << "Position: " << event.order.position << "\n";
out << "User tag: " << (unsigned int)event.order.userTag << "\n\n";
out << "Latency: " << event.latency << "ns\n" << std::endl;
break;
}
case Magmio::TEventType::EVT_REPLACE_SEND: {
out << "REPLACE of #" << event.order.orderID << " SENT\n";
out << "New ID: " << event.order.newOrderID << "\n";
out << "Size: " << event.order.size << "\n";
out << "Symbol: " << event.order.symbol << "\n";
out << "Symbol (num): " << event.order.symbolNum << "\n";
out << "SID: " << event.order.sid << "\n";
out << "User tag: " << (unsigned int)event.order.userTag << "\n\n";
out << "Latency: " << event.latency << "ns\n" << std::endl;
break;
}
case Magmio::TEventType::EVT_CANCEL_SEND: {
out << "CANCEL of Order #" << event.order.orderID << " SENT\n";
out << "New ID: " << event.order.newOrderID << "\n";
out << "User tag: " << (unsigned int)event.order.userTag << "\n\n";
out << "Latency: " << event.latency << "ns\n" << std::endl;
break;
}
case Magmio::TEventType::EVT_ORDER_ACK: {
out << "ORDER #" << event.order.orderID << " ACKED (market ID #" << event.order.marketID << ")\n" << std::endl;
break;
}
case Magmio::TEventType::EVT_CANCEL_ACK: {
out << "CANCEL #" << event.order.orderID << " ACKED\n" << std::endl;
break;
}
case Magmio::TEventType::EVT_ORDER_FILLED: {
out << "ORDER #" << event.order.orderID << " FILLED (increment: " << event.order.size << ")\n" << std::endl;
break;
}
case Magmio::TEventType::EVT_ORDER_KILLED: {
out << "ORDER #" << event.order.orderID << " KILLED (decrement: " << event.order.size << ")\n" << std::endl;
break;
}
case Magmio::TEventType::EVT_DONE_FOR_DAY: {
out << "ORDER #" << event.order.orderID << " IS DONE FOR DAY\n" << std::endl;
break;
}
case Magmio::TEventType::EVT_MESSAGE_REJECTED: {
out << "Message with order id #" << event.order.orderID << " REJECTED\n";
out << "Description: " << event.infoText << "\n" << std::endl;
break;
}
case Magmio::TEventType::EVT_ERROR: {
out << "ERROR EVENT: " << event.infoText << "\n" << std::endl;
return Magmio::OTHER_ERROR;
}
case Magmio::TEventType::EVT_REPLAY_COMPLETE: {
out << "REPLAY COMPLETED" << std::endl;
break;
}
default: {
out << "Unknown Order Event\n" << std::endl;
break;
}
}
return Magmio::SUCCESS;
}
Magmio::TError sendOrderExample(Magmio::MagmioApi &api, const std::string sessionId, const uint64_t sid, const uint64_t initOrderId, const uint16_t count, const unsigned int msDelay) {
Magmio::TError error;
Magmio::TSendOrder swOrder;
uint64_t orderId = initOrderId;
if (!api.order.isRunning(sessionId)) {
std::cerr << "ERROR (Order Sender): Order sender session " << sessionId << " is not running!" << std::endl;
return Magmio::NOT_RUNNING;
}
// Default order values
swOrder.sid = sid;
swOrder.side = Magmio::SIDE_BUY;
swOrder.orderType = Magmio::TYPE_LIMIT;
swOrder.timeInForce = Magmio::TIF_DAY;
swOrder.iso = Magmio::ISO_YES;
swOrder.position = Magmio::POSITION_CLOSE;
swOrder.askPrice = 1000;
swOrder.askSize = 10;
for (int i = 0; i < count; i++) {
if ((error = api.order.sendOrderFpga(sessionId, orderId, swOrder)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Order Sender): Order sending failed - " << Magmio::Log::getErrorMessage(error) << std::endl;
return Magmio::OTHER_ERROR;
} else {
std::cout << "Order Sender: Sw Order " << orderId << " sent on session " << sessionId << std::endl;
}
// Update values for next order
orderId++;
swOrder.askPrice += 100;
swOrder.askSize += 10;
// Delay sending of the next order
std::this_thread::sleep_for(std::chrono::milliseconds(msDelay));
}
return Magmio::SUCCESS;
}
void runOrderSender(std::promise<bool> &&initStatus, std::promise<Magmio::TError> &&exitStatus, std::atomic<bool> &stop, Magmio::MagmioApi &api,
cpu_set_t cpuSet, const uint16_t watchdogMsTimeout) {
Magmio::TError setupError {Magmio::SUCCESS};
Magmio::TError processError {Magmio::SUCCESS};
Magmio::TEventItem event;
std::thread watchdogThread;
std::atomic<bool> stopWatchdog;
stopWatchdog.store(false, std::memory_order_release);
// Set thread's affinity
int retcode;
if ((retcode = pthread_setaffinity_np(pthread_self(), sizeof(cpu_set_t), &cpuSet)) != 0) {
std::cerr << "ERROR (Order Sender): Unable to set desired thread affinity - " << strerror(retcode) << std::endl;
initStatus.set_value(false);
exitStatus.set_value(Magmio::OTHER_ERROR);
return;
}
while (!api.isInitialized()) {
if (stop.load(std::memory_order_acquire)) {
std::cerr << "ERROR (Order Sender): API not initialized!" << std::endl;
initStatus.set_value(false);
exitStatus.set_value(Magmio::NOT_INITIALIZED);
return;
}
std::this_thread::sleep_for(std::chrono::milliseconds(100));
}
// Create file for stream output
std::ofstream outFile("osender.out", std::ofstream::out | std::ofstream::trunc);
if (!outFile.is_open()) {
std::cerr << "ERROR (Order Sender): Output file opening failed!" << std::endl;
initStatus.set_value(false);
exitStatus.set_value(Magmio::OPEN_FILE);
return;
}
// Start order sender
if ((setupError = api.order.start()) != Magmio::SUCCESS) {
std::cerr << "ERROR (Order Sender): Start failed - " << Magmio::Log::getErrorMessage(setupError) << std::endl;
outFile.close();
initStatus.set_value(false);
exitStatus.set_value(std::move(setupError));
return;
} else {
std::cout << "Order Sender: Connected to server." << std::endl;
}
if (watchdogMsTimeout) {
// Enable watchdog in order sender (increase timeout by 2 seconds to ensure that watchdog thread resets before timeout)
api.order.wdt_start(watchdogMsTimeout + 2000);
watchdogThread = std::thread(watchdog, std::ref(stopWatchdog), std::ref(api), watchdogMsTimeout);
}
initStatus.set_value(true);
// Read events from order sender in infinite loop (until stopped or failed)
while (!stop.load(std::memory_order_acquire)) {
// Read event
processError = api.order.getEvent(event);
if (processError == Magmio::RUN_AGAIN) {
continue;
} else if (processError != Magmio::SUCCESS) {
std::cerr << "ERROR (Order Sender): " << Magmio::Log::getErrorMessage(processError) << std::endl;
break;
}
// Process event
if ((processError = processOrderEvent(outFile, event)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Order Sender): Processing event failed!" << std::endl;
break;
}
}
// Stop order sender
if ((setupError = api.order.stop()) != Magmio::SUCCESS) {
std::cerr << "ERROR (Order Sender): Stop failed - " << Magmio::Log::getErrorMessage(setupError) << std::endl;
}
if (watchdogThread.joinable()) {
stopWatchdog.store(true, std::memory_order_release);
watchdogThread.join();
}
outFile.close();
exitStatus.set_value((processError != Magmio::SUCCESS) ? std::move(processError) : std::move(setupError));
}
