97 template <class T> bool AsyncQueue<T>::pop(T &item, int ms) {
98  //lock before reading
99  std::unique_lock<std::mutex> lock(mtx);
100 
101  //if the queue is empty, wait for given time
102  if (ms>0) {
103  cond.wait_for(lock, std::chrono::milliseconds(ms), [this] { return !queue.empty() || interrupted.load(std::memory_order_acquire); });
104  }
105  else if (ms==-1) {
106  cond.wait(lock, [this] { return !queue.empty() || interrupted.load(std::memory_order_acquire); });
107  }
108 
109  //if the queue is empty now, it means timeout
110  if (queue.empty()) {
111  return false;
112  }
113 
114  //data are ready, read them out!
115  item = std::move(queue.front());
116  queue.pop_front();
117 
118  return true;
119 }
