200  bool compare(const ExportTob &other, bool useMemcmp = true) const {
201  if (useMemcmp) {
202  return (*this == other);
203  } else {
204  if (exported == other.exported) {
205  for (unsigned int i = 0; i < exported; i++) {
206  if (!(levels[i] == other.levels[i])) {
207  return false;
208  }
209  }
210  } else {
211  return false;
212  }
213 
214  return true;
215  }
216  }

221  ExportTob() : exported(0), levels(nullptr) {};
222 
223  ExportTob(const ExportTob &other) {
224  if (other.exported && other.levels && (levels = (ExportTobLevel *)malloc(other.exported * sizeof(ExportTobLevel)))) {
225  memcpy(levels, other.levels, other.exported * sizeof(ExportTobLevel));
226  exported = other.exported;

228  levels = nullptr;
