#include <iostream>
#include "global_params.h"
Magmio::TError testGlobalParams(Magmio::MagmioApi &api) {
Magmio::TError error;
Magmio::GlobalParams::tParam param;
Magmio::GlobalParams::tParams params;
const unsigned int globalParamsCount = api.params.global.getNumParams();
if (!api.isInitialized()) {
return Magmio::NOT_INITIALIZED;
}
if (globalParamsCount == 0) {
std::cout << "Global Params: None of global params are available." << std::endl;
return Magmio::SUCCESS;
}
// Set parameter 0 to -1000
std::cout << "Global Params: Set global parameter 0" << std::endl;
if((error = api.params.global.setParam(Magmio::GlobalParams::tParam(0, -1000))) != Magmio::SUCCESS) {
std::cerr << "ERROR (Global Params): Setting parameter 0 failed!" << std::endl;
return error;
}
// Get parameter 0
std::cout << "Global Params: Get global parameter 0" << std::endl;
// Parameter 0
param.first = 0;
// Initialize value to 0. This value will be rewritten
param.second = 0;
if ((error = api.params.global.getParam(param)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Global Params): Getting parameter 0 failed!" << std::endl;
return error;
} else {
std::cout << "Global Params: Parameter 0 = " << (int)param.second << std::endl;
}
// The next tests require at least two global params
if (globalParamsCount < 2) {
return Magmio::SUCCESS;
}
// Set first and last parameters at once (faster)
std::cout << "Global Params: Set first and last global parameters" << std::endl;
params.clear();
params.push_back(Magmio::GlobalParams::tParam(0, 1111));
params.push_back(Magmio::GlobalParams::tParam(globalParamsCount-1, 2222));
if ((error = api.params.global.setParams(params)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Global Params): Setting parameters failed!" << std::endl;
return error;
}
// Get first and last global parameters at once (faster)
std::cout << "Global Params: Get first and last global parameters" << std::endl;
params.clear();
params.push_back(Magmio::GlobalParams::tParam(0, 0));
params.push_back(Magmio::GlobalParams::tParam(globalParamsCount-1, 0));
if ((error = api.params.global.getParams(params)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Global Params): Getting parameters failed!" << std::endl;
return error;
}
for (auto item : params) {
std::cout << "Global Params: Parameter " << (int)item.first << " = " << (int)item.second << std::endl;
}
// Get all parameters at once
std::cout << "Global Params: Get all global parameters at once" << std::endl;
params.clear();
// If no params are specified in vector (empty), all parameters are get
if ((error = api.params.global.getParams(params)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Global Params): Getting all parameters failed!" << std::endl;
return error;
}
for (auto item : params) {
std::cout << "Global Params: Parameter " << (int)item.first << " = " << (int)item.second << std::endl;
}
// Set all paremeters at once
std::cout << "Global Params: Set all global parameters at once" << std::endl;
params.clear();
for (unsigned int i = 0; i < globalParamsCount; i++) {
params.push_back(Magmio::GlobalParams::tParam(i, 1));
}
if ((error = api.params.global.setParams(params)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Global Params): Setting all parameters failed!" << std::endl;
return error;
}
// Get all parameters at once
std::cout << "Global Params: Get all global parameters at once" << std::endl;
params.clear();
// If no params are specified in vector (empty), all parameters are get
if ((error = api.params.global.getParams(params)) != Magmio::SUCCESS) {
std::cerr << "ERROR (Global Params): Getting all parameters failed!" << std::endl;
return error;
}
for (auto item : params) {
std::cout << "Global Params: Parameter " << (int)item.first << " = " << (int)item.second << std::endl;
}
return Magmio::SUCCESS;
}
